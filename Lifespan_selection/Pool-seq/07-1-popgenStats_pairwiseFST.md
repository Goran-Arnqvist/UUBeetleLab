## Script to convert read count format into sync format

To estimate pairwise FST between all possible pool pairs, I will run the R package poolfstat. This package requires population allele freq. data in the synchronized (SYNC) file format, which is the format that is required by [PoPoolation2](https://sourceforge.net/p/popoolation2/wiki/Tutorial/).

Below, there is the custom python script that converts the AD file to the SYNC file format required by poolfstat, `csv2sync.py`:

```python
#!/usr/bin/python3

"""
Created on Jan. 10, 2018. Modified on: Dec. 27, 2019, Uppsala University.
@author: Angela P. Fuentes-Pardo, based on a draft script made by Gavin Douglas. E-mail: apfuentesp@gmail.com.

**Description**:
This script converts a CSV file - with allele depths per sample - into a so called 'synchronized' format, required by Popoolation2

column 3: reference character
column >3: allele frequencies for all populations in the form A-count:T-count:C-count:G-count:N-count:deletion-count

import argparse

def main():

    parser = argparse.ArgumentParser(description="Script that transforms a CSV file with allele per sample into the sync file format required to run Popoolation2",
        formatter_class=argparse.RawDescriptionHelpFormatter)

    parser.add_argument("-i", "--input", metavar="INFILE", type=str,
                        required=True,
                        help="Input CSV with depths.")

    parser.add_argument("-o", "--output", metavar="OUTPUT", type=str,
                        required=True,
                        help="Output sync file.")


    args = parser.parse_args()

    # Start a counter for the number of lines (excluding the header).
    line_count = 0

    # Open the output file for writing.
    outfile = open(args.output, "w")

    # Read through the CSV file line-by-line.
    with open(args.input, "r") as csv_file:
        for line in csv_file:

            # Remove line terminator/newline character.
            line = line.rstrip("\r\n")

            # Split line on tab characters.
            line_split = line.split("\t")

            # Skip if first line.
            if line_count == 0:

                # Save in the list with the headerline the first 4 columns of the header line (CHROM, POS, REF, ALT).
                headerline = line_split[0:3]

                # Append the 1st pop name (5th column) to the header line list, and the other pop names - every column.
                for i in range(4, len(line_split)):
                    headerline.append(line_split[i])

                # Print the header line to the output file.
                print("\t".join(headerline), file=outfile)
                line_count += 1
                continue

            # For each SNP line, do...
            # Print to the output file the columns CHROM, POS, REF.
            print(line_split[0] +  "\t" + line_split[1] + "\t" + line_split[2],
                end="", file=outfile)

            # Loop through each individual sample (i.e. population) = AD (Allele Depth):
            for i in range(4, len(line_split)):

            	# Create a dictionary for each possible allele (A:T:C:G:N:del) per population. Initialize them with zeros.
                nucl_depth = {}
                nucl_depth["A"] = str(0)
                nucl_depth["T"] = str(0)
                nucl_depth["C"] = str(0)
                nucl_depth["G"] = str(0)
                nucl_depth["N"] = str(0)
                nucl_depth["del"] = str(0)

                # Extract the REF and ALT allele read depth of a given population by spliting the string in the comma sign.
                pop_info = line_split[i].split(",")

                # Using the key:value feature of the dictionary, assign the correspondent read depth to the REF (line_split[2]) or ALT alleles (line_split[3]) of a population.
                nucl_depth[line_split[2]] = str(pop_info[0])
                nucl_depth[line_split[3]] = str(pop_info[1])

                # Print the read depth of each possible allele for each population to the output file.
                print("\t" + nucl_depth["A"] + ":" + nucl_depth["T"] + ":" + nucl_depth["C"] + ":" + nucl_depth["G"] + ":" + nucl_depth["N"] + ":" + nucl_depth["del"], end="",
                      file=outfile)

            # Increase the counter after a line is done.
            line_count += 1

            # Make newline.
            print("", file=outfile)

    # Close output file.
    outfile.close()

    # Print in screen the total number of lines (SNPs) processed.
    print("*** Total number of lines processed: [" + str(line_count - 1) + "] ***")


if __name__ == "__main__":
    main()

```

```bash
#!/bin/bash
#SBATCH -A snic2020-15-137
#SBATCH -M snowy
#SBATCH -p core -n 6
#SBATCH -t 10:00:00
#SBATCH -J pwFST
#SBATCH -e pwFST_%J_%A_%a.err
#SBATCH -o pwFST_%J_%A_%a.out
#SBATCH --mail-type=all
#SBATCH --mail-user=angela.fuentespardo@gmail.com
#SBATCH -a 1-3

# Load required programs.
module load bioinfo-tools
module load python3/3.7.2
module load R/3.6.1
module load R_packages/3.6.1

# Set environment variables.
# Files.
WORK_DIR='/proj/snic2020-6-128/private/obtectus_poolseq/users/angela/analysis/07-popgen-stats/pairwiseFST'
INPUT_FILES='/proj/snic2020-6-128/private/obtectus_poolseq/users/angela/analysis/07-popgen-stats/pairwiseFST/list_AD_files.16pools.txt'
POOL_NAMES='EI-F,EI-M,EII-F,EII-M,EIII-F,EIII-M,EIV-F,EIV-M,LV-F,LV-M,LVI-F,LVI-M,LVII-F,LVII-M,LVIII-F,LVIII-M'
POOL_SIZES='96,96,96,96,96,96,96,96,96,96,96,96,96,96,96,96'
MINCOVPERPOOL='10'
MAXCOVPERPOOL='150'
MAF='0.05'
OUTPUTSNPVALUES='FALSE'
COLORLOWHIGH='gray90,gray50' # 'blue,yellow'
POP_ORDER='EI-F,EI-M,EII-F,EII-M,EIII-F,EIII-M,EIV-F,EIV-M,LV-F,LV-M,LVI-F,LVI-M,LVII-F,LVII-M,LVIII-F,LVIII-M'
# Programs.
SYNC_CODE='/proj/snic2020-6-128/private/obtectus_poolseq/users/angela/code/utility-code/csv2sync.py'
R_CODE='/proj/snic2020-6-128/private/obtectus_poolseq/users/angela/code/utility-code/07-1-calculate-pairwiseFST.R'

cd $WORK_DIR

INPUT_target=$(sed -n "$SLURM_ARRAY_TASK_ID"p $INPUT_FILES)
INPUT_PATH=${INPUT_target}
INPUT_FILENAME=$(basename "${INPUT_PATH}")
OUTFILE_PREFIX=$(basename "$INPUT_FILENAME" .txt)
echo This is array job: $SLURM_ARRAY_TASK_ID
echo -e This is the INPUT target: ${INPUT_target}

# Copy required files to the hard drive of the node in Uppmax.
cp $INPUT_PATH* $SNIC_TMP  # With *, the VCF file and its index are copied.

# Go to the SNIC_TMP (working) directory.
cd $SNIC_TMP

# Exclude loci (rows) with NAs from the AD file.
cat $INPUT_FILENAME | grep -v "NA" > noNAs_ADfile

# Transform the file with Neff corrected read counts into a synchronized format.
python3 $SYNC_CODE \
-i noNAs_ADfile \
-o $OUTFILE_PREFIX.sync

# Since the actual sync file format does not have a header, create another file without the header:
cat $OUTFILE_PREFIX.sync | sed '1d' > $OUTFILE_PREFIX.noheader.sync

# Save the header in a separate file.
cat $OUTFILE_PREFIX.sync | head -1 > header_$OUTFILE_PREFIX.sync.txt

# Clean up.
rm noNAs_ADfile
rm $OUTFILE_PREFIX.sync
mv $OUTFILE_PREFIX.noheader.sync $OUTFILE_PREFIX.sync

# Copy resulting files to the working directory.
cp $SNIC_TMP/$OUTFILE_PREFIX.sync $WORK_DIR
cp $SNIC_TMP/header_$OUTFILE_PREFIX.sync.txt $WORK_DIR

INPUTSYNC_FILENAME=$(basename "$INPUT_FILENAME" .txt)

# Run the R script, passing on the working directory and input file path (IN THIS ORDER!!)
Rscript $R_CODE $WORK_DIR $INPUTSYNC_FILENAME.sync $POOL_NAMES $POOL_SIZES $MINCOVPERPOOL $MAXCOVPERPOOL $MAF $OUTPUTSNPVALUES $COLORLOWHIGH $POP_ORDER
```
