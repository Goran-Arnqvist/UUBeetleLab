# 2021-03-31

The main filtering steps I will follow are:

1. **GATK hard-filtering (hf)**: The goal is to obtain a data set containing only variant sites that we trust with high confidence. Then, we will remove variants that have skewed GATK quality annotation values across all samples
2. **DP and GQ**: Filter on minimum and maximum read depth (DP) and genotype quality (GQ)
3. **Monomorphic, missingness, and MAF**: Filter out monomorphic SNPs, samples with a high amount of missing data, variants with a high amount of missing genotypes, and filter on minor allele frequency

I will also apply the filters used by Fan Han for comparison. While I expect the main genomic regions of divergence will remain the same, by using the set of filters I have been using frequently I remove loci with low coverage that can inflate global FST values.

```bash
INPUT_FILE=$(basename "${INPUT_PATH}")
OUTFILE_PREFIX='OTT_16_pools.UG'
# Programs.
GATK_JAR='/sw/apps/bioinfo/GATK/3.8-0/GenomeAnalysisTK.jar'

# Copy required files to the hard drive of the node in Uppmax (SNIC_TMP).
cp $INPUT_PATH* $SNIC_TMP # With *, the vcf file and its index will be copied.
cp $REF_PATH* $SNIC_TMP  # With *, copy the reference and associated files.
cp ${REF_PATH/.fasta/.dict} $SNIC_TMP  # Copy the dict file of the reference. To replace the first occurrence of a pattern with a given string, use ${parameter/pattern/string}

# Go to the SNIC_TMP (working) directory.
cd $SNIC_TMP

# 2.1) Apply GATK hard filters. Flag SNPs that do not pass GATK hard-filters.
java -Djava.io.tmpdir=$SNIC_TMP/tmp -XX:ParallelGCThreads=1 -Dsamjdk.use_async_io=true -Dsamjdk.buffer_size=4194304 -Xmx14G -jar $GATK_JAR \
-T VariantFiltration \
-R $REF_FILE \
-V $INPUT_FILE \
--filterExpression "FS > 60.0" --filterName "FS" \
--filterExpression "SOR > 3.0" --filterName "SOR" \
--filterExpression "MQ < 40.0" --filterName "MQ" \
--filterExpression "MQRankSum < -12.5" --filterName "MQRankSum" \
--filterExpression "ReadPosRankSum < -8.0" --filterName "ReadPosRankSum" \
-o $OUTFILE_PREFIX.SNPs.hfflag.vcf

# Retain only variants that passed these filters (source: https://evodify.com/gatk-in-non-model-organism/).
grep -E '^#|PASS' $OUTFILE_PREFIX.SNPs.hfflag.vcf > $OUTFILE_PREFIX.SNPs.hf.vcf

# 2.2) Apply GQ filter to the FORMAT (genotype) field.
java -Djava.io.tmpdir=$SNIC_TMP/tmp -XX:ParallelGCThreads=1 -Dsamjdk.use_async_io=true -Dsamjdk.buffer_size=4194304 -Xmx14G -jar $GATK_JAR \
-T VariantFiltration \
-R $REF_FILE \
-V $OUTFILE_PREFIX.SNPs.hf.vcf \
--genotypeFilterExpression "GQ < 20" \
--genotypeFilterName "GQGTlt20" \
-o $OUTFILE_PREFIX.SNPs.hf.GQ20.flag.vcf.gz

# Set to no-call (./.) the genotypes filtered in the previous step.
java -Djava.io.tmpdir=$SNIC_TMP/tmp -XX:ParallelGCThreads=1 -Dsamjdk.use_async_io=true -Dsamjdk.buffer_size=4194304 -Xmx14G -jar $GATK_JAR \
-T SelectVariants \
-R $REF_FILE \
-V $OUTFILE_PREFIX.SNPs.hf.GQ20.flag.vcf.gz \
--setFilteredGtToNocall \
-o $OUTFILE_PREFIX.SNPs.hf.GQ20.vcf.gz

# 2.3) Exclude monomorphic loci, loci with missing rate greater than 20%, and loci with minimum 3 counts of the alternative allele.
# Remove monomorphic loci.
bcftools filter -e 'AC==0 || AC==AN' -O z -o $OUTFILE_PREFIX.SNPs.hf.GQ20.mono.vcf.gz $OUTFILE_PREFIX.SNPs.hf.GQ20.vcf.gz

# Obtain missingness stats per pool sample.
bcftools stats -s - $OUTFILE_PREFIX.SNPs.hf.GQ20.mono.vcf.gz | grep -E ^PSC | cut -f3,14 > $OUTFILE_PREFIX.SNPs.hf.GQ20.mono.imiss

# Generate a sample missingness plot.
#R --vanilla < plot_imiss_pools.R

# Filter for 20% missing rate and MAC (minor allele count).
bcftools filter -e 'F_MISSING > 0.2 || MAC < 3' -O z -o $OUTFILE_PREFIX.SNPs.hf.GQ20.mono.miss20.mac3.vcf.gz $OUTFILE_PREFIX.SNPs.hf.GQ20.mono.vcf.gz

# Generate an index for the final VCF file.
tabix -p vcf $OUTFILE_PREFIX.SNPs.hf.GQ20.mono.miss20.mac3.vcf.gz

# 2.4) Extract DP field for each variant in the VCF file of filtered SNPs.
java -Djava.io.tmpdir=$SNIC_TMP/tmp -XX:ParallelGCThreads=1 -Dsamjdk.use_async_io=true -Dsamjdk.buffer_size=4194304 -Xmx14G -jar $GATK_JAR \
-T VariantsToTable \
-R $REF_FILE \
-V $OUTFILE_PREFIX.SNPs.hf.GQ20.mono.miss20.mac3.vcf.gz \
-F CHROM -F POS -GF GT -GF DP \
-o $OUTFILE_PREFIX.SNPs.hf.GQ20.mono.miss20.mac3.DP.table

# Number of SNPs after each filtering step.
echo -e "Number of SNPs in $OUTFILE_PREFIX.SNPs.hf.vcf is":
bcftools view -H $OUTFILE_PREFIX.SNPs.hf.vcf | wc -l

#echo -e "Number of SNPs in $OUTFILE_PREFIX.SNPs.hf.GQ20.vcf.gz is":
#bcftools view -H $OUTFILE_PREFIX.SNPs.hf.GQ20.vcf.gz | wc -l

echo -e "Number of SNPs in $OUTFILE_PREFIX.SNPs.hf.GQ20.mono.vcf.gz is":
bcftools view -H $OUTFILE_PREFIX.SNPs.hf.GQ20.mono.vcf.gz | wc -l

echo -e "Number of SNPs in $OUTFILE_PREFIX.SNPs.hf.GQ20.mono.miss20.mac3.vcf.gz is":
bcftools view -H $OUTFILE_PREFIX.SNPs.hf.GQ20.mono.miss20.mac3.vcf.gz | wc -l

# Copy result files from SNIC_TMP to working directory.
cp $SNIC_TMP/$OUTFILE_PREFIX.SNPs.hf.vcf* $WORK_DIR
cp $SNIC_TMP/$OUTFILE_PREFIX.SNPs.hf.GQ20.mono.imiss $WORK_DIR
cp $SNIC_TMP/$OUTFILE_PREFIX.SNPs.hf.GQ20.mono.miss20.mac3.vcf.gz $WORK_DIR
cp $SNIC_TMP/$OUTFILE_PREFIX.SNPs.hf.GQ20.mono.miss20.mac3.vcf.gz.tbi $WORK_DIR
cp $SNIC_TMP/$OUTFILE_PREFIX.SNPs.hf.GQ20.mono.miss20.mac3.DP.table $WORK_DIR

```

```R
  ## Calculate standard deviation (SD) of depth of coverage per sample.
  library(matrixStats)
  set.seed(42)
  SD <- round(colSds(as.matrix(DPdata_subset), na.rm = FALSE), 1)  # It returns a vector of the standard deviation of the columns.
  SD

  ## Calculate DP interval (Mean +- 1SD) per sample.
  MeanMinus1SD <- Mean-SD
  MeanMinus1SD
  MeanPlus1SD <- Mean+SD
  MeanPlus1SD

  ## Generate a unified df with all the info (mean, SD, -1SD, +1SD).
  DPintervals <- rbind(Mean, SD, MeanMinus1SD, MeanPlus1SD)
  DPintervals
  class(DPintervals)

  # Save this df as a file.   HOM_12_pools.SNPs.hf.vcf
  write.table(DPintervals, file=paste0('Intervals_DPfilter_Mean1SD.',filename,'.',outputName,'.txt'), quote=FALSE, row.names=TRUE, col.names=NA)
}

  rownames(DPintervals) <- c('Mode','Mode1half','ModeMinus1half','ModePlus1half')
  DPintervals

  ## Calculate the mode of the depth of coverage per sample.
  for (i in 1:ncol(DPdata_subset)) {

    DPintervals[c(1),i] <- round(get.mode(DPdata_subset[,i]), 1)  # Mode

    DPintervals[c(3),i] <- round(DPintervals['Mode',i] - DPintervals['Mode1half',i],1)  # Mode - 1/2 mode
    DPintervals[c(4),i] <- round(DPintervals['Mode',i] + DPintervals['Mode1half',i],1)  # Mode + 1/2 mode
  }
  DPintervals
```

```bash
#!/bin/bash
#SBATCH -A snic2020-5-94
#SBATCH -M snowy
#SBATCH -p core -n 4
#SBATCH -t 20:00:00
#SBATCH -J apply_DPfilter
#SBATCH -e apply_DPfilter_%J_%A_%a.err
#SBATCH -o apply_DPfilter_%J_%A_%a.out
#SBATCH --mail-type=all
#SBATCH --mail-user=angela.fuentespardo@gmail.com
#SBATCH --mail-type=ALL
#SBATCH -a 1-3

# Load required programs in the cluster computer.
module load bioinfo-tools
module load vcftools/0.1.15
module load bcftools/1.10
module load samtools/1.10
module load zlib/1.2.11

# Set environment variables.
# Files.
WORK_DIR='/proj/snic2020-6-128/private/obtectus_poolseq/04-vcf-files'
INPUT_FILE='/proj/snic2020-6-128/private/obtectus_poolseq/04-vcf-files/OTT_16_pools.UG.SNPs.hf.GQ20.mono.miss20.mac3.vcf.gz'
SNP_LIST='./list_SNPs_passed_filters.txt'
OUTFILE_PREFIX='./list_output_filenames.txt'

# Go to the SNIC_TMP (working) directory.
cd $WORK_DIR

# Set path to files specific to each job array.
echo This is array job: $SLURM_ARRAY_TASK_ID
SNP_target=$(sed -n "$SLURM_ARRAY_TASK_ID"p $SNP_LIST)
OUTFILE_target=$(sed -n "$SLURM_ARRAY_TASK_ID"p $OUTFILE_PREFIX)
echo -e This is the file listing target SNPs: ${SNP_target}
echo -e This is the output filename: ${OUTFILE_target}

# 1) Apply GATK hard quality filters and FORMAT filters (missing rate, MAC). (Done previously)

# 2) Get VCF with only the SNPs that passed the DP filter.
# Reformat the file listing SNP loci to have one column for chromosome and another one for position, separated by a tab.
awk -F'-' -v OFS='\t' '{print $1,$2}' $SNP_target > $OUTFILE_target.filtered_SNPs.list

# Generate a VCF file with only the filtered SNPs.
vcftools --gzvcf $INPUT_FILE --positions $OUTFILE_target.filtered_SNPs.list --recode --recode-INFO-all --stdout | bgzip > $OUTFILE_target.vcf.gz
# Rename the resulting file (remove the "recode" part)
#mv $OUTFILE_target.recode.vcf $OUTFILE_target.vcf

# Generate an index for the final VCF file.
tabix -p vcf $OUTFILE_target.vcf.gz

# Number of SNPs in the filtered file.
echo -e "Number of SNPs in $OUTFILE_target.vcf.gz is":
bcftools view -H $OUTFILE_target.vcf.gz | wc -l
```
